## Question 2:

```js
function totalKcal(arr) {
  let result = 0;
  // YOUR CODE HERE
  for (i = 0; i < arr.length; i++) {
    result += (arr[i] - 2000) / 7700;
  }

  return result;
}
```

## Question 3:

```js
function totalPrice(arr) {
  let total = 0;
  for (i = 0; i < arr.length; i++) {
    total += arr[i].price;
  }
  return total;
}
```
